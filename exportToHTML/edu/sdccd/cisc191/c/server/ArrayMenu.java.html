55       public void getAtIndex(Scanner scanner) {
56           System.out.print("Enter index: ");
57           int index = scanner.nextInt();
58           if (index >= 0 && index < array.length) {
59               System.out.println("Value at index " + index + ": " + array[index]);
60           } else {
61               System.out.println("Index out of bounds.");
62           }
63       }

65       public void setAtIndex(Scanner scanner) {
66           System.out.print("Enter index: ");
67           int index = scanner.nextInt();
68           System.out.print("Enter value: ");
69           int value = scanner.nextInt();
70           if (index >= 0 && index < array.length) {
71               array[index] = value;
72               System.out.println("Value set at index " + index);
73           } else {
74               System.out.println("Index out of bounds.");
75           }
76       }

78       public void findIndexOf(Scanner scanner) {
79           System.out.print("Enter value: ");
80           int value = scanner.nextInt();
81           for (int i = 0; i < array.length; i++) {
82               if (array[i] == value) {
83                   System.out.println("Value found at index " + i);
84                   return;
85               }
86           }
87           System.out.println("Value not found.");
88       }

90       public void printAll() {
91           for (int i : array) {

93           }

95       }

97       public void deleteAtIndex(Scanner scanner) {
98           System.out.print("Enter index: ");
99           int index = scanner.nextInt();
100          if (index >= 0 && index < array.length) {
101              array[index] = 0; // Assuming 0 is the default "empty" value
102              System.out.println("Value deleted at index " + index);
103          } else {
104              System.out.println("Index out of bounds.");

106      }

108      public void expand() {
